-Like-a-file-

import urllib.request, urllib.parse, urllib.error

fhand = urllib.request.urlopen('http://data.pr4e.org/romeo.txt') # method.method.functionopenfile

counts = dict() # we make a dictionary
for line in fhand: # we read the file  in each line 
    words = line.decode().split() #we add decode  because that line is not a string and we split each line
    for word in words: # we make a loop to the word in each word
        counts[word] = counts.get(word, 0) + 1 # we count each word
print(counts) # and we print each word of the dictionary
